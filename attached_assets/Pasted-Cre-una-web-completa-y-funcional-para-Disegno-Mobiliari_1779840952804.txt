SKILLS GSAP OBLIGATORIAS:
javascriptgsap.registerPlugin(ScrollTrigger, ScrollToPlugin);

// 1. Cursor personalizado
const cursor = document.querySelector('.cursor');
gsap.quickTo(cursor, "x", {duration: 0.3, ease: "power3"});
gsap.quickTo(cursor, "y", {duration: 0.3, ease: "power3"});

// 2. SplitText manual para el hero
const chars = title.textContent.split('').map(char => `<span>${char}</span>`);
gsap.from(chars, { y: -60, opacity: 0, stagger: 0.03, ease: "power4.out" });

// 3. Parallax en imágenes
gsap.to(image, { yPercent: -20, ease: "none",
  scrollTrigger: { trigger: section, scrub: 1 }
});

// 4. Reveal con clipPath
gsap.from(element, { clipPath: "inset(100% 0 0 0)",
  scrollTrigger: { trigger: element, start: "top 80%" }
});

// 5. Horizontal scroll en proyectos
gsap.to(track, { xPercent: -100,
  scrollTrigger: { trigger: container, pin: true, scrub: 1 }
});
